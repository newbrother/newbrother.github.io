## 풀이

시뮬레이션

삼성SW기출문제.

두번째 푸니까 확실히 깔끔하게 잘 푼것같다.

감시구역을 check하는 check()함수를 따로 만들었고,

cctv들을 vector에 저장해둔다음에 cctv개수만큼 모든 경우를 탐색했다.

각cctv마다 탐색해야할 경우 가지수를 cctvType에 저장해뒀다.(4,2,4,4,1)

각 경우마다 해당 방향으로 check를 해준뒤, cnt==cctv.size가 되면 

최소 사각지대를 구하고, ans에 갱신 한다.

check함수에 dir을 보낼때 범위체크를 하기보단,

check함수 내부에서 dir=dir%4를 하면 깔끔하게 check할 수 있다.

## 코드

```cpp
#include <iostream>
#include <queue>
#include <deque>
#include <vector>
#include <cstring>
#include <functional>
using namespace std;

const int cctvType[] = { 0,4,2,4,4,1 };
//1:각방향
//2:좌우, 위아래
//3: 붙어있는 2방향씩
//4: 3방향씩
//5 : 4방향
int n, m;
int arr[8][8];
vector<pair<pair<int, int>, int> > v;

//오른,아래,왼,위
const int dx[] = { 0,1,0,-1 };
const int dy[] = { 1,0,-1,0 };
int ans = 987654321;

void check(int x,int y,int dir) {
    dir = dir % 4;
    int nx = x + dx[dir];
    int ny = y + dy[dir];
    while (1) {
        if (nx >= 0 && nx < n&&ny >= 0 && ny < m&&arr[nx][ny] != 6) {
            if (arr[nx][ny] == 0) {
                arr[nx][ny] = -1;
            }
            nx += dx[dir];
            ny += dy[dir];
        }
        else {
            break;
        }
    }
}

void go(int cnt) {
    if (cnt == v.size()) {
        //최소크기 세고 갱신
        int tmp = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (arr[i][j] == 0)
                    tmp++;
            }
        }
        if (tmp < ans)
            ans = tmp;
        return;
    }
    int backup[8][8];
    memcpy(backup, arr, sizeof(arr));
    for (int i = 0; i < cctvType[v[cnt].second]; i++) {
        int x = v[cnt].first.first;
        int y = v[cnt].first.second;
        if (v[cnt].second == 1) {
            check(x, y, i);
        }
        else if (v[cnt].second == 2) {
            check(x, y, i);
            check(x, y, i+2);
        }
        else if (v[cnt].second == 3) {
            check(x, y, i);
            check(x, y, i+1);
        }
        else if (v[cnt].second == 4) {
            check(x, y, i);
            check(x, y, i+1);
            check(x, y, i+2);
        }
        else if (v[cnt].second == 5) {
            check(x, y, 0);
            check(x, y, 1);
            check(x, y, 2);
            check(x, y, 3);
        }
        go(cnt + 1);
        //rollback
        memcpy(arr, backup, sizeof(arr));
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    //최소 사각지대
    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> arr[i][j];
            if (arr[i][j] >= 1 && arr[i][j] <= 5) {
                v.push_back({ {i,j},arr[i][j] });
            }
        }
    }
    go(0);
    cout << ans;
    return 0;
}
```
